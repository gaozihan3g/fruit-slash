def segment_circle_hit(a, b, center, radius):
    ab = b - a
    if ab.length_squared() == 0:
        return center.distance_to(a) <= radius

    t = clamp(
        (center - a).dot(ab) / ab.length_squared(),
        0, 1
    )
    return center.distance_to(a + ab * t) <= radius
